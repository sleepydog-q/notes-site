import numpy as np

class Tensor:
    def __init__(self, data, parents=(), grad_fn=None, requires_grad=False):
        self.data = np.array(data, dtype=np.float32)
        self.parents = parents          # Tensors used to create this tensor
        self.grad_fn = grad_fn          # backward function (closure)
        self.requires_grad = requires_grad
        self.grad = np.zeros_like(self.data)

    def backward(self, grad=None):
        if grad is None:
            # starting from a scalar loss
            grad = np.ones_like(self.data)
        self.grad += grad
        if self.grad_fn is not None:
            self.grad_fn(grad)

    def __add__(self, other):
        other = other if isinstance(other, Tensor) else Tensor(other)
        out_data = self.data + other.data

        def grad_fn(grad_out):
            if self.requires_grad:
                self.parents[0].backward(grad_out)
            if other.requires_grad:
                self.parents[1].backward(grad_out)

        return Tensor(
            out_data,
            parents=(self, other),
            grad_fn=grad_fn,
            requires_grad=(self.requires_grad or other.requires_grad),
        )

    def matmul(self, other):
        out_data = self.data @ other.data

        def grad_fn(grad_out):
            # dL/dA = grad_out @ B^T
            if self.requires_grad:
                self.parents[0].backward(grad_out @ other.data.T)
            # dL/dB = A^T @ grad_out
            if other.requires_grad:
                self.parents[1].backward(self.data.T @ grad_out)

        return Tensor(
            out_data,
            parents=(self, other),
            grad_fn=grad_fn,
            requires_grad=(self.requires_grad or other.requires_grad),
        )
